# normal function example......
def factorial(n):
    f = 1
    for i in range(1, n + 1):
        f *= i
    return f 

# recursion example
def factorial(n):
    if n > 0:
        return n * factorial(n - 1)
    else:
        return 1

# OR
# recursion example single line
def factorial(n):
    return n * factorial(n - 1) if n > 0 else 1
